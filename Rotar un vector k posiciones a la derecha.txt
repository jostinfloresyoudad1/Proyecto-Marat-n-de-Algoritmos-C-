#include <iostream>
#include <vector>
#include <algorithm> // Para std::rotate (alternativa, pero lo haremos manualmente)


int main() {
int n, k;
// 1. Pedir al usuario el tamaño del vector y sus elementos
std::cout << "Ingresa el tamaño del vector: ";
std::cin >> n;
if (n <= 0) return 1;
std::vector<int> numeros(n);
std::cout << "Ingresa los " << n << " elementos del vector:" << std::endl;
for (int i = 0; i < n; ++i) {
std::cin >> numeros[i];
}
// 2. Pedir al usuario el número de posiciones k a rotar
std::cout << "¿Cuántas posiciones quieres rotar a la derecha? ";
std::cin >> k;
// 3. Ajustar k
k = k % n;
// 4. Crear un vector temporal
std::vector<int> temporal(n);
// 5. Copiar elementos a sus nuevas posiciones
for (int i = 0; i < n; ++i) {
int nuevaPosicion = (i + k) % n;
temporal[nuevaPosicion] = numeros[i];
}
// 6. Copiar de vuelta al vector original
numeros = temporal;
// 7. Mostrar el vector rotado
std::cout << "Vector rotado: ";
for (int num : numeros) {
std::cout << num << " ";
}
std::cout << std::endl;
return 0;
}
